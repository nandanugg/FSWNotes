1. Reverse String
Soal : Mengubah susunan sebuah string dengan memutarbalikkan susunan katanya
Jawaban : 
const word = "Make me a sandwich!"
function reverseString(words) {
    let splitString = words.split("") // Memisahkan setiap huruf menjadi array
    let reverseArray = splitString.reverse(); // Membalikkan urutan array
    let joinArray = reverseArray.join(""); // Menggabungkan array
    return joinArray;
}
console.log(reverseString(words))

Soal : Mengidentifikasi kata yang dibalik sama dengan kata aslinya.

  "Kayak",
  "Kitchen",

  "Madam",
  "Working",
  "Mom",

  "Noon",
  "Racecar",
].map(v => v.toLowerCase()); // Membuat seluruh huruf kecil

function filterPalindromes(list) {
  return words.filter((word) => {
    for(var i = 0; i < word.length / 2; i++) {

    return true;
  });
}
