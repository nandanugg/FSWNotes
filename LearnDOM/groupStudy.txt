1. Reverse string
        membuat suatu kalimat atau kata dibalik urutannya. Misal Kemarin Sore menjadi eroS nirameK
    source code:
const word = "Make me a sandwich!"

function reverse(word){
    var splitWord = word.split("");
    var reverse = splitWord.reverse();
    var join = reverse.join("");
    return join;
}

console.log(reverse(word))
