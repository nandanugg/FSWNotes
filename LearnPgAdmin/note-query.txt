- - kueri buat table dengan constraint (penahan)
CREATE TABLE, user(
	id BIGSERIAL PRIMARY KEY ,
	// PRIMARY KEY, untuk pembeda data satu dg yg lain
		first_name VARCHAR (50) NOT NULL UNIQUE,
		// NOT NULL biar ngisinya tidak kosong
		// UNIQUE biar tidak samadengan siian yg lain
		last_name VARCHAT (50),
		gender VARCHAT (7),
		date_of_birth TIMESTAMP
);

- - create table dengan relasi ke table user
CREATE TABLE pets(
	id BIGSERIAL PRIMARY KEY,
	name VARCHAR  (50) NOR NULL,
	user_id BIGSERIAL,
	FOREIGN KEY(user_id) REFERENCE users(id)
)
