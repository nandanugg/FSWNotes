    console.log(value);

}

console.log(findIndexResult); // 16 adalah index ke 2, index dimulai dari 0

// promise adalah sebuah class yg harus ada imbuhan 'new' diawalnya
function requestDataFromInternet(url, cb) {
    return new Promise((resolve, reject) => { //reject adalah variabel yg menjadi fungsi
        if (url) {
            resolve(url)
        } else {
            reject('url ga ada')
        }
    });
}
