3. Simple function return boolean

membuat fungsi sederhana mengembalikan nilai true jika 2 angka dijumlahkan tidak lebih besar dari 100

function lessThan(num1, num2) {
    const sum = num1 + num2
    if (sum < 100) {
        return true
    } else {
        return false
    }
}
console.log(lessThan(202, 15))

5. simple jquery add comment and delete comment

membuat jquery untuk menambahkan dan menghapus komentar pada halaman website

const input = $('.input')
const buttonKirim = $('.btn-kirim')
const buttonHapus = $('.btn-hapus')
const comment = $('.comment')
buttonKirim.click(() => {
    const p = $('<p></p>')
    p.html(input.val())
    comment.append(p)
    input.val("")
})

// menghapus seluruh elemen p didalam class "comment"
buttonHapus.click(() => {
    $('.comment>p').remove(":contains('" + comment.val() + "')")
})
