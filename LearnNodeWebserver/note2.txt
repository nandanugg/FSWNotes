untuk melakukan route otomatis melalui param yang di inputkan di address bar pada browser
app.get('/:path', (req, res) => {
    res.render(req.params.path, {}, (err, html) => {
        if (err) {
            res.render('404')
        } else {
            res.send(html)
        }
    })
})
